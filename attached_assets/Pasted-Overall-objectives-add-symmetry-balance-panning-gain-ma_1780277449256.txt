import Foundation
import CoreAudio
import Atomics

class CoreAudioInputDeviceManager {
    
    /// Queries the macOS system bus for a list of all currently connected audio device hardware IDs
    static func getConnectedInputDevices() -> [AudioDeviceID] {
        var propertyAddress = AudioObjectPropertyAddress(
            mSelector: kAudioHardwarePropertyDevices,
            mScope: kAudioObjectPropertyScopeGlobal,
            mElement: kAudioObjectPropertyElementMain
        )
        
        var dataSize: UInt32 = 0
        var status = AudioObjectGetPropertyDataSize(AudioObjectID(kAudioObjectSystemObject), &propertyAddress, 0, nil, &dataSize)
        guard status == noErr else { return [] }
        
        let deviceCount = Int(dataSize) / MemoryLayout<AudioDeviceID>.size
        var deviceIDs = [AudioDeviceID](repeating: 0, count: deviceCount)
        
        status = AudioObjectGetPropertyData(AudioObjectID(kAudioObjectSystemObject), &propertyAddress, 0, nil, &dataSize, &deviceIDs)
        guard status == noErr else { return [] }
        
        // Filter out devices containing exactly 0 input channels (keep microphones exclusively)
        return deviceIDs.filter { deviceID in
            return getInputChannelCount(deviceID: deviceID) > 0
        }
    }
    
    private static func getInputChannelCount(deviceID: AudioDeviceID) -> Int {
        var propertyAddress = AudioObjectPropertyAddress(
            mSelector: kAudioDevicePropertyStreamConfiguration,
            mScope: kAudioObjectPropertyScopeInput,
            mElement: kAudioObjectPropertyElementMain
        )
        
        var dataSize: UInt32 = 0
        var status = AudioObjectGetPropertyDataSize(deviceID, &propertyAddress, 0, nil, &dataSize)
        guard status == noErr else { return 0 }
        
        let bufferListPointer = UnsafeMutableRawPointer.allocate(byteCount: Int(dataSize), alignment: MemoryLayout<AudioBufferList>.alignment)
        defer { bufferListPointer.deallocate() }
        
        status = AudioObjectGetPropertyData(deviceID, &propertyAddress, 0, nil, &dataSize, bufferListPointer)
        guard status == noErr else { return 0 }
        
        let bufferList = bufferListPointer.assumingMemoryBound(to: AudioBufferList.self).pointee
        let buffers = UnsafeBufferPointer(start: &bufferList.mBuffers, count: Int(bufferList.mNumberBuffers))
        
        return buffers.reduce(0) { $0 + Int($1.mNumberChannels) }
    }

2. Isolated Real-Time Input Recording IO Loop
Instruct the agent to instantiate an independent AudioComponentInstance input unit (kAudioUnitType_Output/ kAudioUnitSubType_HALOutput) specifically configured to execute a real-time recording loop on an isolated thread, completely separated from your main AURemoteIO output stream path to maintain absolute digital isolation.

import CoreAudio
import AudioToolbox
import Atomics

class RoomMeasurementRecordingEngine {
    
    private var inputAudioUnit: AudioUnit? = nil
    private var isRecording = ManagedAtomic<Bool>(false)
    
    // Lock-free pre-allocated static storage memory pool arrays matching your SIMD boundaries
    private var wetCaptureBuffer = [Float]()
    private let maxCaptureSamples = 384000 * 6 // Allocation runtime limit ceiling (6 seconds @ 384kHz max)
    
    init() {
        wetCaptureBuffer.reserveCapacity(maxCaptureSamples)
    }
    
    /// Initializes and links the input audio unit thread directly to your USB microphone hardware ID
    func initializeInputChannelHook(microphoneDeviceID: AudioDeviceID) -> OSStatus {
        var desc = AudioComponentDescription(
            componentType: kAudioUnitType_Output,
            componentSubType: kAudioUnitSubType_HALOutput, // Direct HAL access node connection
            componentManufacturer: kAudioUnitManufacturer_Apple,
            componentFlags: 0,
            componentFlagsMask: 0
        )
        
        guard let component = AudioComponentFindNext(nil, &desc) else {
            return kAudioUnitErr_CannotDoInCurrentContext
        }
        
        var status = AudioComponentInstanceNew(component, &inputAudioUnit)
        guard status == noErr, let au = inputAudioUnit else { return status }
        
        // 1. Enable Input IO on Bus 1, and disable Output IO on Bus 0
        var enableInput: UInt32 = 1
        status = AudioUnitSetProperty(au, kAudioOutputUnitProperty_EnableIO, kAudioUnitScope_Input, 1, &enableInput, UInt32(MemoryLayout<UInt32>.size))
        guard status == noErr else { return status }
        
        var disableOutput: UInt32 = 0
        status = AudioUnitSetProperty(au, kAudioOutputUnitProperty_EnableIO, kAudioUnitScope_Output, 0, &disableOutput, UInt32(MemoryLayout<UInt32>.size))
        guard status == noErr else { return status }
        
        // 2. Bind the selected USB microphone device ID directly to this unit


        var targetDevice = microphoneDeviceID
        status = AudioUnitSetProperty(au, kAudioOutputUnitProperty_CurrentDevice, kAudioUnitScope_Global, 0, &targetDevice, UInt32(MemoryLayout<AudioDeviceID>.size))
        guard status == noErr else { return status }
        
        // 3. Enforce our strict 32-bit floating-point linear PCM stream constraints
        var streamFormat = AudioStreamBasicDescription(
            mSampleRate: 44100.0, // Will automatically scale via auto-clock hooks up to 384kHz
            mFormatID: kAudioFormatLinearPCM,
            mFormatFlags: kAudioFormatFlagIsFloat | kAudioFormatFlagIsPacked | kAudioFormatFlagIsNonInterleaved,
            mBytesPerPacket: 4,
            mFramesPerPacket: 1,
            mBytesPerFrame: 4,
            mChannelsPerFrame: 1, // Capture room readings in tightly focused omni-mono
            mBitsPerChannel: 32,
            mReserved: 0
        )
        
        status = AudioUnitSetProperty(au, kAudioUnitProperty_StreamFormat, kAudioUnitScope_Output, 1, &streamFormat, UInt32(MemoryLayout<AudioStreamBasicDescription>.size))
        guard status == noErr else { return status }
        
        // 4. Hook the hardware render input recording callback container block
        var callbackStruct = AURenderCallbackStruct(
            inputProc: recordingInputCallbackProc,
            inputProcRefCon: UnsafeMutableRawPointer(Unmanaged.passUnretained(self).toOpaque())
        )
        
        status = AudioUnitSetProperty(au, kAudioOutputUnitProperty_SetInputCallback, kAudioUnitScope_Global, 0, &callbackStruct, UInt32(MemoryLayout<AURenderCallbackStruct>.size))
        guard status == noErr else { return status }
        
        status = AudioUnitInitialize(au)
        return status
    }
    
    func startCapture() {
        guard let au = inputAudioUnit else { return }
        wetCaptureBuffer.removeAll(keepingCapacity: true)
        isRecording.store(true, ordering: .relaxed)
        AudioOutputUnitStart(au)
    }
    
    func stopCapture() -> [Float] {
        guard let au = inputAudioUnit else { return [] }
        AudioOutputUnitStop(au)
        isRecording.store(false, ordering: .relaxed)
        return wetCaptureBuffer
    }
}

### 3. Real-Time Hardware Render Recording Input Callback Hook

This low-level global C-style callback intercepts the incoming audio chunks from the USB microphone hardware abstraction bus layer instantly, writing sample arrays into your static pre-allocated memory pool without dynamic memory allocations or thread-stalling blocks.

```swift
private func recordingInputCallbackProc(
    inRefCon: UnsafeMutableRawPointer,
    ioActionFlags: UnsafeMutablePointer<AudioUnitRenderActionFlags>,
    inTimeStamp: UnsafePointer<AudioTimeStamp>,
    inBusNumber: UInt32,
    inNumberFrames: UInt32,
    ioData: UnsafeMutablePointer<AudioBufferList>?
) -> OSStatus {
    
    let engine = Unmanaged<RoomMeasurementRecordingEngine>.fromOpaque(inRefCon).takeUnretainedValue()
    guard let au = engine.inputAudioUnit else { return noErr }
    
    // Allocate a temporary stack buffer list matching our omni-mono constraints
    var bufferList = AudioBufferList()
    bufferList.mNumberBuffers = 1
    bufferList.mBuffers.mNumberChannels = 1
    bufferList.mBuffers.mDataByteSize = inNumberFrames * 4
    bufferList.mBuffers.mData = UnsafeMutableRawPointer.allocate(byteCount: Int(bufferList.mBuffers.mDataByteSize), alignment: 16)
    defer { bufferList.mBuffers.mData?.deallocate() }
    
    // Force the input hardware layer to pull raw PCM measurements down into our buffer list
    let status = AudioUnitRender(au, ioActionFlags, inTimeStamp, inBusNumber, inNumberFrames, &bufferList)
    guard status == noErr else { return status }
    
    if engine.isRecording.load(ordering: .relaxed) {
        if let rawData = bufferList.mBuffers.mData {
            let floatSamples = rawData.assumingMemoryBound(to: Float.self)
            // Append array slices directly inside our pre-allocated memory pool without triggering malloc locks
            for frame in 0..<Int(inNumberFrames) {
                if engine.wetCaptureBuffer.count < engine.maxCaptureSamples {
                    engine.wetCaptureBuffer.append(floatSamples[frame])
                }
            }
        }
    }
    
    return noErr
}
```
