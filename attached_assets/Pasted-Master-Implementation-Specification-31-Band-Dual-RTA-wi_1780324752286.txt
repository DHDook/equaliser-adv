Master Implementation Specification: 31-Band Dual RTA with Target Curve, Custom Gradients, Hover Tooltips, and Realigned Master Meters
Context & Objective
We are completely restructuring the user interface and metering pipeline of the equaliser-adv application (extending the architecture found in DHDook/equaliser-adv and cvknage/equaliser). The existing vertical master meters inside /meters compute values over time-domain buffers and are structurally separated from the frequency view.
This specification merges all metering, DSP analysis, thread safety, and custom UI modules into a single, cohesive vertical layout container.
Core Layout Restructuring:
Top Zone (Master Meters): The existing legacy Peak In/Out and RMS In/Out master meters must be converted into a space-efficient, horizontal progress bar layout spanning left-to-right (\(0\%\) to \(100\%\) scaling). This row features an integrated, glowing hardware-style clip indicator light with an automatic 1.5-second hold timer and an interactive click-to-reset gesture.
Bottom Zone (Spectral RTA Graph Window): Positioned directly below the master meters, a dual-view 31-Band Real-Time Spectrum Analyzer (RTA) will render both Input (Pre-EQ) and Output (Post-EQ) energy, overlaid with a continuous, vector-drawn Target EQ Filter Curve.
Interactive Tooltip Engine: Hovering the cursor over any of the 31 columns must instantly generate a floating, pixel-aligned informational HUD box tracking the precise center frequency, raw Input/Output decibels, and a color-coded Delta Gain (\(\Delta\text{ dB}\)) Readout.
FPS UI Performance Monitor: To ensure real-time calculations do not compromise thread fluidity, a toggleable diagnostics panel must track and render a continuous frame rate counter.
Global Gain, Sample Rate, and Bypass Rules:
Audiophile-Grade Tracking (Up to 384 kHz): The data pipeline must process any hardware sample rate up to 384 kHz dynamically. The 31-band grouping limits must scale fluidly based on the reported system clock rate using a lock-free, mono-summed stereo ring buffer architecture.
Global Gain Offsets: Global input preamp gain (in dB) must visually offset the Input RTA graph. Global output gain (in dB) must offset the Output RTA graph and accumulate directly into the vertical offset calculation of the Target EQ Line.
Bypass Isolation Rule: When the master isEqBypassed flag is true, only the Output RTA bars, Output Peak Ticks, the canvas measurement grid, and the Target EQ Line will change to a desaturated, faded, or flattened \(0\text{ dB}\) state. The Input RTA bars and Input Peak Ticks must continue rendering in full color normally.

struct BandData {
    var currentValue: Float = -60.0
    var peakValue: Float = -60.0
    var peakHoldFrames: Int = 0
}

struct BiquadCoefficients {
    let b0: Double, b1: Double, b2: Double
    let a1: Double, a2: Double
}

class LockFreeAudioRingBuffer {
    private var buffer: [Float]
    private let bufferSize = 8192
    private var writeIndex: Int = 0
    
    init() {
        self.buffer = [Float](repeating: -60.0, count: bufferSize)
    }
    
    /// Called by real-time audio thread. Sums stereo to mono via vDSP and writes lock-free.
    func writeStereoSamples(leftChannel: UnsafePointer<Float>, rightChannel: UnsafePointer<Float>, frameCount: Int) {
        var monoFrames = [Float](repeating: 0.0, count: frameCount)
        vDSP_vadd(leftChannel, 1, rightChannel, 1, &monoFrames, 1, vDSP_Length(frameCount))
        var scale: Float = 0.5
        vDSP_vsmul(monoFrames, 1, &scale, &monoFrames, 1, vDSP_Length(frameCount))
        
        for i in 0..<frameCount {
            let targetIdx = (writeIndex + i) % bufferSize
            buffer[targetIdx] = monoFrames[i]
        }
        writeIndex = (writeIndex + frameCount) % bufferSize
    }
    
    /// Extract a contiguous block backward from the write head
    func readLatestChunk(size: Int = 2048) -> [Float] {
        let currentWriteIdx = writeIndex
        var chunk = [Float](repeating: -60.0, count: size)
        var startIdx = currentWriteIdx - size
        if startIdx < 0 { startIdx += bufferSize }
        
        for i in 0..<size {
            let sourceIdx = (startIdx + i) % bufferSize
            chunk[i] = buffer[sourceIdx]
        }
        return chunk
    }
}

class AdvancedDualSpectrumAnalyzer: ObservableObject {
    let minDb: Float = -60.0
    let maxDb: Float = 12.0
    
    private let fftSize = 2048
    private var fftSetup: FFTSetup
    private var log2n: vDSP_Length
    private var window: [Float]
    
    // Analog Ballistics Constants (Emulating IEC 60268-10 PPM Meters at 60 FPS)
    private let decayConstant: Float = 0.33      // ~20 dB drop per second
    private let peakHoldDuration = 30           // Hold peak line static for 500ms
    private let peakDecayConstant: Float = 0.66  // ~40 dB drop per second after hold
    
    let centerFrequencies: [Float] = [
        20, 25, 31.5, 40, 50, 63, 80, 100, 125, 160, 200, 250, 315, 400, 500, 630, 
        800, 1000, 1250, 1600, 2000, 2500, 3150, 4000, 5000, 6300, 8000, 10000, 12500, 16000, 20000
    ]
    
    @Published var inputBands = [BandData](repeating: BandData(), count: 31)
    @Published var outputBands = [BandData](repeating: BandData(), count: 31)
    @Published var targetLinePoints = [Float](repeating: 0.0, count: 31)
    
    @Published var showInputPeaks: Bool = true
    @Published var showOutputPeaks: Bool = true
    
    init() {
        self.log2n = vDSP_Length(log2(Double(fftSize)))
        self.fftSetup = vDSP_create_fftsetup(log2n, FFTRadix(kFFTRadix2))!
        self.window = [Float](repeating: 0, count: fftSize)
        vDSP_hann_window(&window, vDSP_Length(fftSize), Int32(vDSP_HANN_NORM))
    }
    
    func normalizeDb(_ db: Float) -> Float {
        let clamped = max(minDb, min(maxDb, db))
        return (clamped - minDb) / (maxDb - minDb)
    }
    
    /// Evaluates the theoretical biquad frequency responses H(z) mapping directly to the RTA normalization scale
    func updateTargetLine(activeFilters: [BiquadCoefficients], outputGainDb: Float, sampleRate: Double) {
        let points: [Float] = centerFrequencies.map { fc in
            var accumulatedGainDb: Double = Double(outputGainDb)
            let omega = 2.0 * Double.pi * Double(fc) / sampleRate
            
            let cosW = cos(omega), cos2W = cos(2.0 * omega)
            let sinW = sin(omega), sin2W = sin(2.0 * omega)
            
            for filter in activeFilters {
                let numReal = filter.b0 + filter.b1 * cosW + filter.b2 * cos2W
                let numImag = -(filter.b1 * sinW + filter.b2 * sin2W)
                let denReal = 1.0 + filter.a1 * cosW + filter.a2 * cos2W
                let denImag = -(filter.a1 * sinW + filter.a2 * sin2W)
                
                let numMagSq = (numReal * numReal) + (numImag * numImag)
                let denMagSq = (denReal * denReal) + (denImag * denImag)
                
                if denMagSq > 0 {
                    accumulatedGainDb += 10.0 * log10(numMagSq / denMagSq)
                }
            }
            return normalizeDb(Float(accumulatedGainDb))
        }
        
        DispatchQueue.main.async {
            self.targetLinePoints = points
        }
    }
    
    func updateSmearedSpectrums(inputSamples: [Float], inputGainDb: Float, outputSamples: [Float], outputGainDb: Float, sampleRate: Float) {
        var rawInputDb = executeFFT(samples: inputSamples)
        var rawOutputDb = executeFFT(samples: outputSamples)
        
        if inputGainDb != 0 { vDSP_vsadd(&rawInputDb, 1, &inputGainDb, &rawInputDb, 1, vDSP_Length(rawInputDb.count)) }
        if outputGainDb != 0 { vDSP_vsadd(&rawOutputDb, 1, &outputGainDb, &rawOutputDb, 1, vDSP_Length(rawOutputDb.count)) }
        
        let targetInputBands = mapBinsToBands(dbMagnitudes: rawInputDb, sampleRate: sampleRate)
        let targetOutputBands = mapBinsToBands(dbMagnitudes: rawOutputDb, sampleRate: sampleRate)
        
        DispatchQueue.main.async {
            self.applyPhysics(currentBands: &self.inputBands, targetDbValues: targetInputBands)
            self.applyPhysics(currentBands: &self.outputBands, targetDbValues: targetOutputBands)
        }
    }
    
    private func executeFFT(samples: [Float]) -> [Float] {
        guard samples.count >= fftSize else { return [Float](repeating: minDb, count: fftSize / 2) }
        var windowedSamples = [Float](repeating: 0, count: fftSize)
        vDSP_vmul(samples, 1, window, 1, &windowedSamples, 1, vDSP_Length(fftSize))
        
        var real = [Float](repeating: 0, count: fftSize / 2), imag = [Float](repeating: 0, count: fftSize / 2)
        var splitComplex = DSPSplitComplex(realp: &real, imagp: &imag)
        
        windowedSamples.withUnsafeBytes { rawBuffer in
            let typedBuffer = rawBuffer.bindMemory(to: DSPComplex.self)
            vDSP_ctoz(typedBuffer.baseAddress!, 2, &splitComplex, 1, vDSP_Length(fftSize / 2))
        }
        
        vDSP_fft_zrip(fftSetup, &splitComplex, 1, log2n, FFTDirection(FFT_FORWARD))
        var magnitudes = [Float](repeating: 0, count: fftSize / 2)
        vDSP_zvmags(&splitComplex, 1, &magnitudes, 1, vDSP_Length(fftSize / 2))
        
        var scalingFactor = Float(1.0 / Float(fftSize))
        var dbMagnitudes = [Float](repeating: 0, count: fftSize / 2)
        vDSP_vdbcon(&magnitudes, 1, &scalingFactor, &dbMagnitudes, 1, vDSP_Length(fftSize / 2), 1)
        return dbMagnitudes
    }
    
    private func mapBinsToBands(dbMagnitudes: [Float], sampleRate: Float) -> [Float] {
        var mappedBands = [Float](repeating: minDb, count: 31)
        let binWidth = sampleRate / Float(fftSize)
        
        for k in 0..<31 {
            let fc = centerFrequencies[k]
            let lower = fc * pow(2.0, -1.0/6.0)
            let upper = fc * pow(2.0, 1.0/6.0)
            
            var sum: Float = 0.0
            var count: Float = 0.0
            
            for i in 0..<(fftSize / 2) {
                let freq = Float(i) * binWidth
                if freq >= lower && freq <= upper {
                    sum += dbMagnitudes[i]
                    count += 1.0
                }
            }
            mappedBands[k] = count > 0 ? (sum / count) : dbMagnitudes[max(0, min(Int(round(fc / binWidth)), (fftSize / 2) - 1))]
        }
        return mappedBands
    }
    
    private func applyPhysics(currentBands: inout [BandData], targetDbValues: [Float]) {
        for i in 0..<31 {
            let target = targetDbValues[i]
            if target > currentBands[i].currentValue {
                currentBands[i].currentValue = target
            } else {
                currentBands[i].currentValue = max(target, currentBands[i].currentValue - decayConstant)
            }
            if target >= currentBands[i].peakValue {
                currentBands[i].peakValue = target
                currentBands[i].peakHoldFrames = peakHoldDuration
            } else {
                if currentBands[i].peakHoldFrames > 0 {
                    currentBands[i].peakHoldFrames -= 1
                } else {
                    currentBands[i].peakValue = max(minDb, currentBands[i].peakValue - peakDecayConstant)
                }
            }
        }
    }
    deinit { vDSP_destroy_fftsetup(fftSetup) }
}

class MasterMeterStore: ObservableObject {
    @Published var inputPeakFraction: Float = 0.0
    @Published var inputRmsFraction: Float = 0.0
    @Published var outputPeakFraction: Float = 0.0
    @Published var outputRmsFraction: Float = 0.0
    
    @Published var inputIsClipping: Bool = false
    @Published var outputIsClipping: Bool = false
    
    private let clipHoldFramesThreshold = 90 // 1.5 seconds at 60 FPS
    private var inputClipTimerCount = 0
    private var outputClipTimerCount = 0
    
    @Published var showDiagnostics: Bool = false
    @Published var currentFps: Int = 60
    private var lastFpsTimestamp = Date()
    private var frameCountSinceLastFpsUpdate = 0
    
    /// Synchronizes macro levels and tracks clipping hold windows sequentially
    func updateMasterMeters(rawInPeak: Float, rawInRms: Float, rawOutPeak: Float, rawOutRms: Float) {
        DispatchQueue.main.async {
            self.inputPeakFraction = rawInPeak
            self.inputRmsFraction = rawInRms
            self.outputPeakFraction = rawOutPeak
            self.outputRmsFraction = rawOutRms
            
            // Input Clip Tracking
            if rawInPeak >= 1.0 {
                self.inputIsClipping = true
                self.inputClipTimerCount = self.clipHoldFramesThreshold
            } else if self.inputIsClipping {
                self.inputClipTimerCount -= 1
                if self.inputClipTimerCount <= 0 { self.inputIsClipping = false }
            }
            
            // Output Clip Tracking
            if rawOutPeak >= 1.0 {
                self.outputIsClipping = true
                self.outputClipTimerCount = self.clipHoldFramesThreshold
            } else if self.outputIsClipping {
                self.outputClipTimerCount -= 1
                if self.outputClipTimerCount <= 0 { self.outputIsClipping = false }
            }
            
            self.calculateFpsDiagnostics()
        }
    }
    
    func manuallyResetClips() {
        inputIsClipping = false
        inputClipTimerCount = 0
        outputIsClipping = false
        outputClipTimerCount = 0
    }
    
    private func calculateFpsDiagnostics() {
        frameCountSinceLastFpsUpdate += 1
        let now = Date()
        let elapsed = now.timeIntervalSince(lastFpsTimestamp)
        if elapsed >= 1.0 {
            self.currentFps = Int(Double(frameCountSinceLastFpsUpdate) / elapsed)
            self.frameCountSinceLastFpsUpdate = 0
            self.lastFpsTimestamp = now
        }
    }
}
